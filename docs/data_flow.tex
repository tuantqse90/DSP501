\caption{10-Fold Cross-Validation với predefined folds (không shuffle)}
\label{fig:10fold-cv}

Công thức tính 95\% CI:
\begin{equation}
    \text{CI}_{95\%} = \bar{x} \pm 1.96 \cdot \frac{s}{\sqrt{N}}, \quad
    \text{trong đó } \bar{x} = \frac{1}{N}\sum_{i=1}^{N} x_i, \quad
    s = \sqrt{\frac{1}{N-1}\sum_{i=1}^{N}(x_i - \bar{x})^2}
\end{equation}

\subsection{Công thức DSP trong Pipeline B}

\paragraph{FIR Bandpass Filter} sử dụng phương pháp cửa sổ (Hann window):
\begin{equation}
    y[n] = \sum_{k=0}^{M} h[k] \cdot x[n-k], \quad M = 100 \text{ (order 101)}
\end{equation}
Áp dụng zero-phase filtering (\texttt{filtfilt}): lọc xuôi rồi lọc ngược để triệt tiêu phase shift.

Đáp ứng tần số lý tưởng:
\begin{equation}
    |H(f)| = \begin{cases}
        1 & \text{nếu } f_L \leq |f| \leq f_H \\
        0 & \text{ngược lại}
    \end{cases}, \quad f_L = 50\text{\,Hz}, \quad f_H = 10000\text{\,Hz}
\end{equation}

\paragraph{Pre-emphasis Filter} — Bộ lọc thông cao bậc 1:
\begin{equation}
    y[n] = x[n] - \alpha \cdot x[n-1], \quad \alpha = 0.97
\end{equation}
Đáp ứng tần số: $H(z) = 1 - \alpha z^{-1}$, tăng $\sim$20\,dB cho tần số cao.

\paragraph{Peak Normalization:}
\begin{equation}
    \hat{y}[n] = \frac{y[n]}{\displaystyle\max_{n} |y[n]|}
\end{equation}

\subsection{Tại sao chọn FIR thay vì IIR?}

\begin{table}[H]
\centering
\caption{So sánh FIR và IIR filter}
\label{tab:fir-vs-iir}
\begin{tabular}{@{}lcc@{}}
\toprule
\textbf{Thuộc tính} & \textbf{FIR (order=101)} & \textbf{IIR Butterworth (order=5)} \\
\midrule
Pha              & Tuyến tính (linear phase)      & Phi tuyến (nonlinear phase) \\
Ổn định          & Luôn ổn định                    & Cần kiểm tra ($|p_i| < 1$) \\
Bảo toàn dạng sóng & Có                           & Không (phase distortion) \\
Số hệ số         & 101 taps                        & 11 taps ($b_0\ldots b_5$, $a_0\ldots a_5$) \\
Transition band  & Rộng hơn                        & Hẹp hơn \\
\bottomrule
\end{tabular}
\end{table}

\textbf{Lý do chọn FIR:} Các class non-stationary (gun\_shot, dog\_bark) có thông tin quan trọng ở \emph{hình dạng xung} (temporal shape). FIR với pha tuyến tính bảo toàn hình dạng này, trong khi IIR gây méo do phase distortion.

\subsection{Công thức MFCC}

\textbf{Bước 1:} Mel spectrogram — áp dụng mel filterbank lên power spectrum:
\begin{equation}
    S_\text{mel}(m, t) = \sum_{k} |X(k, t)|^2 \cdot W_m(k), \quad m = 1, \ldots, 128
\end{equation}
trong đó $W_m(k)$ là mel filter thứ $m$, $X(k,t)$ là STFT.

Thang mel chuyển đổi tần số $f$ (Hz) sang mel:
\begin{equation}
    m = 2595 \cdot \log_{10}\!\left(1 + \frac{f}{700}\right)
\end{equation}

\textbf{Bước 2:} MFCC — áp dụng DCT lên log mel spectrogram:
\begin{equation}
    c_i(t) = \sum_{m=1}^{M} \log\bigl(S_\text{mel}(m, t)\bigr) \cdot \cos\!\left[\frac{\pi i}{M}\left(m - \frac{1}{2}\right)\right], \quad i = 1, \ldots, 40
\end{equation}

\textbf{Bước 3:} Delta (đạo hàm bậc 1) và Delta-Delta (bậc 2):
\begin{equation}
    \Delta c_i(t) = \frac{\sum_{\theta=1}^{\Theta} \theta \bigl(c_i(t+\theta) - c_i(t-\theta)\bigr)}{2\sum_{\theta=1}^{\Theta} \theta^2}
\end{equation}

\subsection{7 Statistical Aggregation Functions}

Mỗi feature sequence $\{v_1, v_2, \ldots, v_T\}$ (gồm $T$ frames) được nén thành 7 con số cố định:

\begin{align}
    \mu &= \frac{1}{T}\sum_{t=1}^{T} v_t & & \text{(mean — giá trị trung bình)} \\[4pt]
    \sigma &= \sqrt{\frac{1}{T-1}\sum_{t=1}^{T}(v_t - \mu)^2} & & \text{(std — độ biến thiên)} \\[4pt]
    v_\text{min} &= \min_{t} v_t & & \text{(min — giá trị nhỏ nhất)} \\[4pt]
    v_\text{max} &= \max_{t} v_t & & \text{(max — giá trị lớn nhất)} \\[4pt]
    \tilde{v} &= \text{median}(v_1, \ldots, v_T) & & \text{(median — trung vị)} \\[4pt]
    \gamma_1 &= \frac{1}{T}\sum_{t=1}^{T}\!\left(\frac{v_t - \mu}{\sigma}\right)^{\!3} & & \text{(skewness — độ lệch)} \\[4pt]
    \gamma_2 &= \frac{1}{T}\sum_{t=1}^{T}\!\left(\frac{v_t - \mu}{\sigma}\right)^{\!4} - 3 & & \text{(kurtosis — độ nhọn)}
\end{align}

\textbf{Tại sao cần 7 stats?} Hai class có thể có $\mu$ giống nhau nhưng $\sigma$, $\gamma_2$ khác nhau. Ví dụ: \texttt{air\_conditioner} và \texttt{engine\_idling} có mean MFCC tương tự nhưng std khác biệt rõ.

\subsection{Công thức kiểm định thống kê}

\paragraph{Paired $t$-test:}
\begin{equation}
    t = \frac{\bar{d}}{s_d / \sqrt{N}}, \quad
    \bar{d} = \frac{1}{N}\sum_{i=1}^{N} d_i, \quad
    s_d = \sqrt{\frac{1}{N-1}\sum_{i=1}^{N}(d_i - \bar{d})^2}, \quad
    d_i = b_i - a_i
\end{equation}
với $N = 10$ folds, $a_i$ = accuracy Pipeline A tại fold $i$, $b_i$ = accuracy Pipeline B.

\paragraph{Cohen's $d$ effect size:}
\begin{equation}
    d = \frac{\bar{d}}{s_d}, \quad
    \text{Interpretation: } |d| < 0.2 \text{ (negligible)}, \quad
    0.2 \leq |d| < 0.5 \text{ (small)}, \quad
    0.5 \leq |d| < 0.8 \text{ (medium)}
\end{equation}
